\begin{frame}
\frametitle{Maze Mapping Strategy}
\begin{columns}
\column{0.4\textwidth}
\begin{itemize}
    \item \textbf{Exploration Phase:} Robot discovers maze structure
    \item \textbf{Mapping:} Build internal representation using a 2D grid
    \item \textbf{No Diagonal Movement:} Restrict to orthogonal moves
    \item \textbf{Optimization:} Calculate shortest path after full exploration
\end{itemize}
